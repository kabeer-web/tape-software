import { createContext, useState, useEffect, useCallback } from 'react';
import { getInventory, addInventory, updateInventory, deleteInventory } from '../../api';

export const StockContext = createContext();

const JAMBO_CATS = ['Clear','Tan','Cloth','Masking','Tissue','SuperYellow','SuperClear','Color','Foam'];

export const StockProvider = ({ children }) => {
  const [inventory, setInventory] = useState([]);
  const [loading, setLoading] = useState(true);

  const refreshInventory = useCallback(async () => {
    try {
      const data = await getInventory();
      setInventory(data);
    } catch (err) {
      console.error('Fetch error:', err);
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => { refreshInventory(); }, [refreshInventory]);

  // Logic: Generate Next Roll Number
  const generateNextRollNo = () => {
    const jambos = inventory.filter(i => JAMBO_CATS.includes(i.category));
    if (jambos.length === 0) return "001";
    const rollNumbers = jambos.map(i => parseInt(i.rollNo || i.roll_no || 0)).filter(n => !isNaN(n));
    const max = Math.max(...rollNumbers, 0);
    return String(max + 1).padStart(3, '0');
  };

  const addRoll = async (item) => {
    let finalItem = { ...item };
    
    // Auto Generate Roll No if it's a Jambo
    if (JAMBO_CATS.includes(item.category)) {
      finalItem.rollNo = generateNextRollNo();
    }

    try {
      const saved = await addInventory(finalItem);
      setInventory(prev => [saved, ...prev]);
      return saved;
    } catch (err) {
      console.error("Add item error:", err);
      throw err;
    }
  };

  const adjustStock = async (originalItem, field, delta) => {
    if (!originalItem) return;
    const existing = inventory.find(i => i._id === originalItem._id);
    const currentValue = existing ? Number(existing[field] || 0) : 0;
    const newValue = Number((currentValue + delta).toFixed(2));

    try {
      if (newValue <= 0) {
        if (existing) {
          await deleteInventory(originalItem._id);
          setInventory(prev => prev.filter(i => i._id !== originalItem._id));
        }
      } else {
        if (!existing) {
          const { _id, id, created_at, ...rest } = originalItem;
          const restoredItem = { ...rest, [field]: newValue };
          const saved = await addInventory(restoredItem);
          setInventory(prev => [saved, ...prev]);
        } else {
          const updated = await updateInventory(originalItem._id, { [field]: newValue });
          setInventory(prev => prev.map(i => i._id === originalItem._id ? { ...i, ...updated } : i));
        }
      }
    } catch (err) {
      console.error("Stock adjustment failed:", err);
      throw err;
    }
  };

  const removeItem = async (id) => {
    await deleteInventory(id);
    setInventory(prev => prev.filter(i => i._id !== id));
  };

  return (
    <StockContext.Provider value={{
      inventory, loading, adjustStock, addRoll, removeItem, refreshInventory, setInventory
    }}>
      {children}
    </StockContext.Provider>
  );
};

import { supabase } from './supabase';

// ─── INVENTORY ───────────────────────────────────────────
export const getInventory = async () => {
  const { data, error } = await supabase
    .from('inventory')
    .select('*')
    .order('created_at', { ascending: false });
  if (error) throw new Error(error.message);
  return data.map(i => ({ 
    ...i, 
    _id: i.id, 
    rollNo: i.roll_no, 
    cartonType: i.carton_type 
  }));
};

export const addInventory = async (item) => {
  const payload = {
    category:    item.category,
    type:        item.type || item.category,
    date:        item.date,
    roll_no:     item.rollNo || item.roll_no || null,
    yards:       item.yards ? Number(item.yards) : 0,
    micron:      item.micron || null,
    width:       item.width || null,
    color:       item.color || null,
    brand:       item.brand || null,
    side:        item.side || null,
    ply:         item.ply || null,
    qty:         item.qty ? Number(item.qty) : 0,
    carton_type: item.cartonType || item.type || item.category,
    size:        item.size || null,
  };

  const { data, error } = await supabase
    .from('inventory')
    .insert([payload])
    .select()
    .single();
  if (error) throw new Error(error.message);
  return { ...data, _id: data.id, rollNo: data.roll_no };
};

export const updateInventory = async (id, item) => {
  const updateData = {};
  if (item.yards !== undefined)  updateData.yards = Number(item.yards);
  if (item.qty !== undefined)    updateData.qty   = Number(item.qty);
  if (item.color !== undefined)  updateData.color = item.color;
  if (item.micron !== undefined) updateData.micron = item.micron;
  if (item.width !== undefined)  updateData.width = item.width;

  const { data, error } = await supabase
    .from('inventory')
    .update(updateData)
    .eq('id', id)
    .select()
    .single();
  if (error) throw new Error(error.message);
  return { ...data, _id: data.id, rollNo: data.roll_no };
};

export const deleteInventory = async (id) => {
  const { error } = await supabase.from('inventory').delete().eq('id', id);
  if (error) throw new Error(error.message);
  return true;
};

function App() {
  return (
    <AuthProvider>
      <StockProvider>
        <AccountsProvider>
          <Router>
            <Routes>
              <Route path="/login" element={<PublicRoute />} />
              <Route path="/*" element={
                <ProtectedRoute>
                  <AppLayout />
                </ProtectedRoute>
              } />
            </Routes>
          </Router>
        </AccountsProvider>
      </StockProvider>
    </AuthProvider>
  );
}

// Agar already logged in ho to /login pe nahi jaega
const PublicRoute = () => {
  const { user, loading } = useAuth();
  if (loading) return (
    <div className="min-h-screen bg-[#070707] flex items-center justify-center">
      <div className="w-10 h-10 border-2 border-[#22c55e]/30 border-t-[#22c55e] rounded-full animate-spin" />
    </div>
  );
  if (user) return <Navigate to="/" replace />;
  return <Login />;
};

export default App;
